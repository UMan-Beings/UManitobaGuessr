title Game Configuration & Location Guessing
entryspacing 0.6
loop 
==Game Configuration==

participant Frontend
participant Backend
participant Database

Frontend->Backend:POST /api/v1/games {maxTimerSeconds, totalRounds}

==Location Guessing==

loop 
Backend->Database:Get Random Image
Backend<-Database:Image URL, Image ID, latitude, longitude
Frontend<-Backend:Game State (GUESS), Image URL, Image ID, Game ID
Frontend->Backend:POST /api/v1/games/:gameId/guess {latitude, longitude}
Frontend<-Backend:Game State (REVEAL), Score, True Latitude, True Longitude, Guess Time
Frontend->Backend:POST /api/v1/games/:gameId/next
end
Frontend<-Backend:Game State (FINISHED), Total Score, Average Guess Time
Backend->Database:Save Results to Account

end
