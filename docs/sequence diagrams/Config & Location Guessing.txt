title Game Configuration & Location Guessing
entryspacing 0.6
    participant Frontend
    participant Backend
    participant Database
    Frontend->Backend:POST /api/v1/games {totalRounds, maxTimerSeconds, userId}
    Backend->Database:Create game row
    Backend->Database:Create round rows with location details
    loop
        Frontend<-Backend:Phase(GUESS) and game state for round (first location image, time limit, round#, etc.)
        Frontend->Backend:POST /api/v1/games/{gameId}/guess {x, y, guessTimeSeconds}
        Backend->Database:Add x, y, and time from guess to round row
        alt More Rounds Remaining
            Frontend<-Backend:Phase(REVEAL) and round details (score, true lat and long, guess time)
            Frontend->Backend:POST /api/v1/games/{gameId}/next
        else Game finished
            Frontend<-Backend:Phase(FINISHED), and final cumulative score
        end
    end
